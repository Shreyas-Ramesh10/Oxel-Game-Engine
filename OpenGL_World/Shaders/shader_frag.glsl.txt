#version 330

in vec4 vCol;
in vec2 TexCoord0;
in vec3 Normal;			
																
out vec4 colour;

struct DirectionalLight
{
	vec3 color;
	float ambientIntensity;
	vec3 direction;
	float diffuseIntensity;
};

uniform sampler2D theTexture;		
uniform DirectionalLight directionalLight;	
																
void main()																					
{													
	vec4 ambientColor = vec4(directionalLight.color, 1.0f) * directionalLight.ambientIntensity;
	
	//A.B = |A|.|B| cos(angle) -> after normalzing |A||B| = 1, so -> A.B = cos(angle)
	float diffuseFactor = max(dot(normalize(Normal), normalize(directionalLight.direction)), 0.0f); //Taking max to keep the value above zero.
	vec4 diffuseColor = vec4(directionalLight.color, 1.0f) * directionalLight.diffuseIntensity * diffuseFactor;
	
	colour = texture(theTexture, TexCoord0) * (ambientColor + diffuseColor);
}
